      }
    };

    function setLang(lang) {
      document.querySelectorAll('[data-i18n]').forEach((el) => {
        const key = el.dataset.i18n;
        if (translations[lang] && translations[lang][key]) {
          el.innerHTML = translations[lang][key];
        }
      });
    }

    function filterJobs(country) {
      document.querySelectorAll('.job-card').forEach((card) => {
        const shouldShow = country === 'all' || card.dataset.country === country;
        card.classList.toggle('hide', !shouldShow);
      });
    }

    function addJob() {
      const country = document.getElementById('adminCountry').value.trim();
      const flag = document.getElementById('adminFlag').value.trim() || '🌍';
      const category = document.getElementById('adminCategory').value;
      const salary = document.getElementById('adminSalary').value.trim();
      const work = document.getElementById('adminWork').value.trim();
      const housing = document.getElementById('adminHousing').value.trim();

      if (!country || !work || !salary) {
        alert('Заполните страну, работы и зарплату');
        return;
      }

      const jobs = JSON.parse(localStorage.getItem('mirPowerJobs') || '[]');
      jobs.push({ country, flag, category, salary, work, housing });
      localStorage.setItem('mirPowerJobs', JSON.stringify(jobs));

      renderAddedJobs();

      document.getElementById('adminCountry').value = '';
      document.getElementById('adminFlag').value = '';
      document.getElementById('adminSalary').value = '';
      document.getElementById('adminWork').value = '';
      document.getElementById('adminHousing').value = '';

      alert('Вакансия добавлена');
    }

    function renderAddedJobs() {
      const jobsList = document.getElementById('jobsList');
      if (!jobsList) return;

      document.querySelectorAll('.added-job').forEach((el) => el.remove());
      const jobs = JSON.parse(localStorage.getItem('mirPowerJobs') || '[]');

      jobs.forEach((job) => {
        const card = document.createElement('div');
        card.className = 'info-card job-card added-job';
        card.dataset.country = job.category || 'other';
        card.innerHTML = `
          <div class="icon">${job.flag}</div>
          <h3>${job.country}</h3>
          <div class="job-meta">
            <div><b>Работы:</b> ${job.work}</div>
            <div><b>Зарплата:</b> ${job.salary}</div>
            <div><b>Жильё:</b> ${job.housing || 'уточняется'}</div>
          </div>
          <br />
          <a class="btn btn-primary" href="#operators">Оставить заявку</a>
        `;
        jobsList.appendChild(card);
      });
    }

    function clearAddedJobs() {
      if (confirm('Удалить добавленные вакансии?')) {
        localStorage.removeItem('mirPowerJobs');
        renderAddedJobs();
      }
    }

    window.addEventListener('load', renderAddedJobs);
